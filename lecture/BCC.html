const int maxn = 1e5+5;
int deep[maxn],low[maxn];
void dfs(int a,int pa){
    low[a]=deep[a]=deep[pa]+1;
    for(int x:edge[a])if(x!=pa){
        if(low[x])
            low[a]=min(deep[x],low[a]);
        else
            dfs(x,a);
        low[pa]=min(low[a],low[pa]);
    } 
    if(low[a]>=deep[pa]&&a!=0)
        cout<<"find:"<<pa<<' '<<a<<endl;
}
