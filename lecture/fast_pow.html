    int ans=1;
    while(b){
        if(b&1)
            ans*=a;
        a*=a;
        b>>=1;

矩陣乘法

mat muti(mat a,mat b){
    mat ret;
    rep(i,2) rep(j,2)
        ret.arr[i][j]=0;//歸零
    rep(i,2) rep(j,2) rep(k,2)//矩陣乘法
        ret.arr[j][i]+=a.arr[k][i]*b.arr[j][k];
    return ret;
}

void mypow(mat a,int b){
